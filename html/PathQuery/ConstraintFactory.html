517:             case args.length 
518:             when 2
519:                 parameters = {:path => args[0], :op => args[1]}
520:             when 3

523:                 elsif LoopConstraint.valid_ops.include?(args[1])
524:                     parameters = {:path => args[0], :op => args[1], :loopPath => args[2]}
525:                 else
526:                     parameters = {:path => args[0], :op => args[1], :value => args[2]}
527:                 end
528:             when 4
529:                 parameters = {:path => args[0], :op => args[1], :value => args[2], :extra_value => args[3]}
530:             else
531:                 parameters = args.first
532:             end
533:

535:             suitable_classes = @classes.select { |cls| 
536:                 is_suitable = true
537:                 attr_keys.each { |key| 
538:                     is_suitable = is_suitable && (cls.method_defined?(key)) 
539:                     if key.to_s == "op"
540:                         is_suitable = is_suitable && cls.valid_ops.include?(parameters[key])
541:                     end
542:                 }
543:                 is_suitable
544:             }
545:             if suitable_classes.size > 1
546:                 raise ArgumentError, "More than one class found for #{parameters}"
547:             elsif suitable_classes.size < 1
548:                 raise ArgumentError, "No suitable classes found for #{parameters}"
549:             end
550: 
551:             cls = suitable_classes.first
552:             con = cls.new
553:             parameters.each_pair { |key, value|
554:                 if key == :path || key == :loopPath
555:                     value = Path.new(@query.add_prefix(value), @query.model, @query.subclasses)
556:                 end
557:                 if key == :sub_class
558:                     value = Path.new(value, @query.model)
559:                 end
560:                 con.send(key.to_s + '=', value)
561:             }
562:             con.validate
563:             if con.respond_to?(:code)
564:                 code = con.code
565:                 if code.nil?
566:                     con.code = @query.next_code
567:                 else

569:                     unless Query.is_valid_code(code)
570:                         raise ArgumentError, "Coded must be between A and Z, got: #{code}"
571:                     end

573:                         con.code = @query.next_code
574:                     end
575:                 end
